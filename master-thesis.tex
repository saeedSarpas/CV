  \textbf{Discreteness effects in N-body simulations of cosmological structure
  formation} & \multirow{5}{*}{ECD:~Mar 2017} \\
  \desc{\AIfA} & \\
  \desc{Supervisor: Prof.\ Dr.\ Cristiano Porciani} & \\

  \sep

  \desc{Cosmological N-body simulations compute the gravitational
  interactions between discrete particles instead of solving the Vlasov-Poisson
  equation. This might lead to some nonphysical consequences
  (e.g.~generation of spurious structures, noisy force calculation,
  etc.).}& \\
  \desc{Our approach to study these effects is,} \\
  \initem{1.~generate initial conditions with different number of particles,}\\
  \initem{2.~run N-body simulation,}\\
  \initem{3.~find matching halos between different simulations,}\\
  \initem{4.~study statistical properties of halos along the history of
  the simulations.}\\
\end{longtable}
